location ~* /(source|other)$ {
    deny all;
    return 404;
}
#去掉index.php
location / {
  if (!-e $request_filename){
    rewrite ^(.*)$ /index.php/$1 break;
  }
}

location ~* ^/(public/js/kindeditor/attached)/.*\.(php|php5|js|htm|html)$ {
    deny all;
    return 404;
}
